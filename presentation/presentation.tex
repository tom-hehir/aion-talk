  \begin{itemize}\small
    \item \textbf{Emergent Physical Understanding} — solves non-trivial scientific tasks with a simple linear head on learned representations
    \item \textbf{Superior Performance in the Low-Data Regime} — competitive results with orders of magnitude less data than supervised counterparts
    \item \textbf{Flexible Data Fusion} — arbitrary combinations of observations; cross-modal conditional generation
    \item \textbf{Physical Structure of the Latent Space} — embedding space organizes along physically meaningful directions; powerful rare-object retrieval

  \begin{itemize}\small
    \item Token-based multimodal foundation models: 300M–3B parameters
    \item Bespoke tokenization for heterogeneous multi-survey astronomical data
    \item Competitive to state-of-the-art with simple probing; significantly outperforms supervised baselines in the low-data regime
  \end{itemize}

    \end{column}
    \begin{column}{0.45\textwidth}
      \begin{minipage}[t][0.82\textheight][s]{\linewidth}
        \vfill
        \textbf{Encoder token embedding}
        \begin{equation*}
          \mathbf{e}_t^{(\mathrm{enc})} = \mathrm{Embed}_i(x_t^i) + \mathbf{m}_i + \mathbf{p}_t
        \end{equation*}
        \par\vfill
        \textbf{Decoder token embedding}
        \begin{equation*}
          \mathbf{e}_t^{(\mathrm{dec})} = \mathbf{m}_i + \mathbf{p}_t
        \end{equation*}
        \par\vfill
        {\small
          $\mathbf{m}_i$: learnable modality embedding, one per modality
          \emph{and} survey (encodes instrument provenance).\\[0.4em]
          $\mathbf{p}_t$: positional embedding.
        }
        \par\vfill\mbox{}
      \end{minipage}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{Tokenization: tokenizer objective}
  \textbf{Image \& spectrum flux loss} — inverse-variance-weighted Gaussian NLL,
  leveraging known per-pixel noise from survey pipelines:
  \begin{equation*}
    \mathcal{L}_\mathrm{NLL}
    = \sum_i \frac{1}{2}
      \left\|
        \boldsymbol{\Sigma}_i^{-\frac{1}{2}}\,\mathbf{m}_i
        \!\left(\mathbf{x}_i - \underbrace{\mathrm{Dec}_\theta\!\left(\mathrm{Enc}_\phi(\mathbf{x}_i)\right)}_{\hat{\mathbf{x}}_i}\right)
      \right\|_2^2
  \end{equation*}
  \begin{alertblock}{Spectrum tokenizer adds a mask loss}
    Binary cross-entropy between predicted reliability map
    $\hat{\mathbf{m}}(\lambda)$ and ground-truth sky/quality mask
    $\mathbf{m}(\lambda)$
  \end{alertblock}
\end{frame}

\begin{frame}{Scalar tokenizer}
  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      \centering
      \includegraphics[width=\linewidth]{scalar-quantization-binning.png}
    \end{column}
    \begin{column}{0.49\textwidth}
      \begin{alertblock}{Architecture}
        \small
        Map each scalar to a standard normal variate via the empirical CDF:
        \begin{equation*}
          z_i \;=\; \Phi^{-1}\!\bigl(F_x(x_i)\bigr)
        \end{equation*}
        \vspace{-0.3em}
        Fixed centroids at equally spaced normal quantiles:
        \begin{equation*}
          c_k \;=\; \Phi^{-1}\!\!\left(\frac{k - \tfrac{1}{2}}{K}\right), \quad K = 1024
        \end{equation*}
        \vspace{-0.3em}
        Reconstruction inverts the mapping:
        \begin{equation*}
          \hat{x}_i \;=\; F_x^{-1}\!\bigl(\Phi(c_i)\bigr)
        \end{equation*}
        \vspace{-0.2em}
        No parameters learned; no loss required. Equal probability
        mass per bin adapts automatically to long tails.
      \end{alertblock}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{Pre-training objective: masked token modelling}
  \begin{center}
    \begin{align*}
      \mathcal{L}(\theta)
      &= -H(f, p_\theta) \\
      &= -\mathbb{E}_{(\mathbf{x}^{\mathrm{targ}},\,\mathbf{x}^{\mathrm{cond}})\sim f}
          \!\left[\log p_\theta(\mathbf{x}^{\mathrm{targ}} \mid \mathbf{x}^{\mathrm{cond}})\right] \\
      &\simeq -\sum_{i=1}^{N} \log p_\theta\!\left(\mathbf{x}_i^{\mathrm{targ}} \mid \mathbf{x}_i^{\mathrm{cond}}\right)
    \end{align*}
  \end{center}
  \vspace{0.3em}
  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      \textbf{Input} (budget: 256 tokens)
      \begin{enumerate}\small
        \item Select primary modality at random
        \item Uniformly sample a \# of tokens $n$
        \item Randomly sample $n$ tokens from that modality
        \item Fill remainder by randomly sampling tokens from all other modalities
      \end{enumerate}
    \end{column}
    \begin{column}{0.48\textwidth}
      \textbf{Output} (budget: 128 tokens)
      \begin{enumerate}\small
        \item Select primary modality at random (may differ from input)
        \item Sample \# of tokens $n$ via Beta distribution (skewed low)
        \item Randomly sample $n$ tokens from that modality
        \item Fill remainder by randomly sampling tokens from all other modalities
      \end{enumerate}
    \end{column}
  \end{columns}
\end{frame}

% ── §8.5 Embedding Extraction ─────────────────────────────────────────────────
\begin{frame}{Pooling embeddings}
  \textbf{Protocol}: freeze the encoder; pool the contextualized output sequence
  $\mathbf{Z}=\{\mathbf{z}_1,\dots,\mathbf{z}_T\},\;\mathbf{z}_t\in\mathbb{R}^d$
  to a single object-level vector $\mathbf{e}\in\mathbb{R}^d$.\\[0.8em]
  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      \textbf{Mean pooling}
      \begin{equation*}
        \mathbf{e} \;=\; \frac{1}{T}\sum_{t=1}^{T}\mathbf{z}_t
      \end{equation*}
    \end{column}
    \begin{column}{0.48\textwidth}
      \textbf{Attentive pooling}
      \begin{equation*}
        \mathbf{A} = \mathrm{softmax}\!\Bigl(\tfrac{\mathbf{Q}\mathbf{K}^\top}{\sqrt{d}}\Bigr),\quad
        \mathbf{e} = \mathbf{A}\mathbf{V}
      \end{equation*}
      {\small ($\mathbf{Q},\mathbf{K},\mathbf{V}$ are learnable projections)}
    \end{column}
  \end{columns}
\end{frame}

  Conclusions

\begin{frame}{Future work}
  \textbf{Architecture}
  \begin{itemize}
    \item Replace quantised tokens with \textbf{continuous tokens} to lift
          the information ceiling imposed by codebook discretisation
    \item Better decoder: current masked-token decoder does not produce a
          proper conditional distribution given the encoder embedding —
          candidates include autoregressive modelling, diffusion-based
          decoding, or dropping decoding entirely in favour of a
          \textbf{JEPA-style} objective
    \item Better pooling: mean pooling is suboptimal; consider
          self-supervised discriminative post-training for
          task-aware summary embeddings
  \end{itemize}
  \vspace{0.5em}
  \textbf{Community adoption}
  \begin{itemize}
    \item Integration into survey pipelines as a source of ML-inferred
          data products
    \item Best use-cases today: \textbf{embeddings} for parameter estimation
          and \textbf{similarity retrieval} — generation not yet recommended
          for production use
  \end{itemize}
\end{frame}

% ── §11 Conclusions ───────────────────────────────────────────────────────────
\begin{frame}{Conclusions}
  \begin{itemize}
    \item \textbf{Integrates} imaging, spectroscopy, and scalar data across
          five large-scale surveys via modality-specific tokenisation and a
          shared masked-modelling objective
    \item \textbf{Achieves or surpasses state-of-the-art} on galaxy and
          stellar parameter estimation, morphology classification, image
          segmentation, and spectral super-resolution
    \item \textbf{Excels in the low-data regime} — competitive performance
          with $10^2$–$10^3$ labels, two to three orders of magnitude fewer
          than fully supervised baselines
    \item \textbf{Enables zero-shot semantic retrieval} of rare objects
          (mergers, strong gravitational lenses) without task-specific labels
    \item \textbf{Enables cross-survey transfer}: representations are largely
          agnostic to instrument-specific characteristics
    \item A scalable, open-source blueprint for multimodal foundation models
          in other scientific domains
  \end{itemize}
\end{frame}

% ── Appendix: Subsampled Linear Projection ───────────────────────────────────
\begin{frame}{Subsampled linear projection}
  \textbf{Multi-survey image tokenizer}: 9-channel input (5 HSC $+$ 4 LS bands)
  projected to a ${\approx}6C = 54$-dimensional space:
  \begin{equation*}
    \hat{\mathbf{x}} \;=\; \alpha(\mathbf{m}_c)\bigl(\tilde{\mathbf{x}}W + b\bigr),
    \qquad
    W\!\in\!\mathbb{R}^{C\times\mathrm{dim}_\mathrm{out}},\;
    b\!\in\!\mathbb{R}^{\mathrm{dim}_\mathrm{out}}
  \end{equation*}
  \begin{alertblock}{Purpose}
    Disentangles survey-specific channel information into a higher-dimensional embedding.
    The scale factor $\alpha(\mathbf{m}_c)$ keeps the feature norm invariant to missing
    channels (i.e.\ when a survey provides fewer than 9 bands).
    Projection is inverted after decoding.
  \end{alertblock}
  \vspace{0.3em}
  The 54-dimensional output feeds into the ResNet (MagViT-style) autoencoder.
\end{frame}

% ── Appendix: ROAR Sampling ───────────────────────────────────────────────────
\begin{frame}{Inference sampling: ROAR}
  At inference time, draw posterior samples of any modality via iterative token resampling:
  \begin{equation*}
    p_\theta\!\bigl(\mathbf{x}^{\mathrm{qry}} \mid \mathbf{x}^{\mathrm{in}}\bigr)
    \;=\;
    \prod_{j\in\mathcal{Q}}
    p_\theta\!\bigl(x_j \mid \mathbf{x}^{\mathrm{in}}\bigr)
  \end{equation*}
  \textbf{ROAR generation} \parencite{mizrahi2023}: at each iteration $t$:
  \begin{enumerate}\small
    \item Draw a random permutation $\pi_t\!:\mathcal{Q}_t\!\to\!\mathcal{Q}_t$ of the unknown query indices
    \item Reveal the first $\rho_t = \lfloor r^{\,t}|\mathcal{Q}_t|\rfloor$ positions:
          $\;\mathcal{S}_t = \{\pi_t(1),\dots,\pi_t(\rho_t)\}$
    \item Sample those tokens:
          $\;x_j^{(t)} \sim p_\theta\!\bigl(\cdot \mid \mathbf{x}^{\mathrm{in}} \cup
          \mathbf{x}^{(t-1)}_{\mathcal{Q}_{t-1}\setminus\mathcal{S}_t}\bigr),\; j\in\mathcal{S}_t$
    \item Promote to inputs and update:
          $\;\mathbf{x}^{\mathrm{in}} \leftarrow \mathbf{x}^{\mathrm{in}} \cup \{x_j^{(t)}\}_{j\in\mathcal{S}_t}$,
          $\;\mathcal{Q}_{t+1} = \mathcal{Q}_t\setminus\mathcal{S}_t$
  \end{enumerate}
  With decay $r\in[0,1)$, full sample generated in $T=\mathcal{O}(\log|\mathcal{Q}|)$ decoder calls.
\end{frame}
