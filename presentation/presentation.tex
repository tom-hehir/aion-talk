% ── §7 Architecture & Objective ───────────────────────────────────────────────
\begin{frame}{Architecture: Encoder–Decoder Transformer}
  \begin{columns}[T]
    \begin{column}{0.52\textwidth}
      \centering
      \includegraphics[width=\linewidth, height=0.78\textheight, keepaspectratio]{aion.png}
    \end{column}
    \begin{column}{0.45\textwidth}
      \textbf{Encoder token embedding}
      \begin{equation*}
        \mathbf{e}_t^{(\mathrm{enc})} = \mathrm{Embed}_i(x_t^i) + \mathbf{m}_i + \mathbf{p}_t
      \end{equation*}
      \textbf{Decoder token embedding}
      \begin{equation*}
        \mathbf{e}_t^{(\mathrm{dec})} = \mathbf{m}_i + \mathbf{p}_t
      \end{equation*}
      {\small
        $\mathbf{m}_i$: learnable modality embedding, one per modality
        \emph{and} survey (encodes instrument provenance).\\[0.4em]
        $\mathbf{p}_t$: positional embedding.\\[0.4em]
        Decoder omits $\mathrm{Embed}_i(x_t^i)$ — it receives only
        modality $+$ position, not the token value.
      }
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{Tokenizer Training: Negative Log-Likelihood}
  \textbf{Image \& spectrum flux loss} — inverse-variance-weighted Gaussian NLL,
  leveraging known per-pixel noise from survey pipelines:
  \begin{equation*}
    \mathcal{L}_\mathrm{NLL}
    = \sum_i \frac{1}{2}
      \left\|
        \mathbf{\Sigma}_i^{-\frac{1}{2}}\,\mathbf{m}_i
        \!\left(\mathbf{x}_i - \mathrm{Dec}_\theta\!\left(\mathrm{Enc}_\phi(\mathbf{x}_i)\right)\right)
      \right\|_2^2
  \end{equation*}
  \vspace{0.4em}
  \textbf{Spectrum tokenizer adds a mask loss:}
  \begin{itemize}
    \item Binary cross-entropy between predicted reliability map
          $\hat{\mathbf{m}}(\lambda)$ and ground-truth sky/quality mask
          $\mathbf{m}(\lambda)$
    \item Likelihood-based, but no principled way to weight against the
          flux NLL
  \end{itemize}
\end{frame}

\begin{frame}{Pre-Training Objective: Masked Token Modelling}
  \begin{center}
    \begin{align*}
      \mathcal{L}(\theta)
      &= -H(f, p_\theta) \\
      &= -\mathbb{E}_{(\mathbf{x}^{\mathrm{targ}},\,\mathbf{x}^{\mathrm{cond}})\sim f}
          \!\left[\log p_\theta(\mathbf{x}^{\mathrm{targ}} \mid \mathbf{x}^{\mathrm{cond}})\right] \\
      &\simeq -\sum_{i=1}^{N} \log p_\theta\!\left(\mathbf{x}_i^{\mathrm{targ}} \mid \mathbf{x}_i^{\mathrm{cond}}\right)
    \end{align*}
  \end{center}
  \vspace{0.3em}
  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      \textbf{Input} (budget: 256 tokens)
      \begin{enumerate}\small
        \item Select primary modality at random
        \item Uniformly sample a \# of tokens $n$
        \item Randomly sample $n$ tokens from that modality
        \item Fill remainder by randomly sampling tokens from all other modalities
      \end{enumerate}
    \end{column}
    \begin{column}{0.48\textwidth}
      \textbf{Output} (budget: 128 tokens)
      \begin{enumerate}\small
        \item Select primary modality at random (may differ from input)
        \item Sample \# of tokens $n$ via Beta distribution (skewed low)
        \item Randomly sample $n$ tokens from that modality
        \item Fill remainder by randomly sampling tokens from all other modalities
      \end{enumerate}
    \end{column}
  \end{columns}
  \vspace{0.4em}
  Discretisation is necessary: cross-entropy over discrete tokens gives a
  \textbf{proper probabilistic objective} — continuous latents do not.
\end{frame}
